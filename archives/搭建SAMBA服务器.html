Centos7搭建SAMBA服务器

安装：yum -y install samba
创建文件夹和用户
创建文件夹：# mkdir smbuser 创建用户：# useradd -s /sbin/nologin smbuser
创建samba用户：

smbpasswd -a smbuser
New SMB password:
Retype new SMB password:
Added user logadmin.

更改目录属性：

#chown smbuser.smbuser smbuser   
#chmod -R 777 smbuser

配置samba服务：

vi /etc/samba/smb.conf
[shared]
    # 共享文件目录描述
    comment = Shared Directories
    # 共享文件目录
    path = /storage/shared/
    # 是否允许guest访问
    public = no
    # 指定管理用户
    admin users = admin
    # 可访问的用户组、用户
    valid users = @admin
    # 是否浏览权限
    browseable = yes
    # 是否可写权限
    writable = yes
    # 文件权限设置
    create mask = 0777
    directory mask = 0777
    force directory mode = 0777
    force create mode = 0777

保存退出，重启samba。
